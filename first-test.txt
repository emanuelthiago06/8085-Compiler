VALOR:      DB 4
MVI B,4          ;B = 4
            MVI C,2          ;C = 2
            MVI D,0          ;D = 0
            MVI E,0          ;E = 0
            MOV A,B        ;A = B
LOOP: INR D              ;D++
            SUB C             ;A = A - C
            JP LOOP         ;if(A >= 0) goto LOOP
            ADD C            ;A = A + C
            DCR D            ;D--
            MOV E,A       ;E = A
            HLT
